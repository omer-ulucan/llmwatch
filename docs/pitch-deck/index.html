Programmatic Access

One API key. Every feature. Same pattern as Stripe & OpenAI.

# Chat Completion
curl http://your-host:3000/api/chat/completions \
  -H "X-API-Key: lw_your_key_here" \
  -H "Content-Type: application/json" \
  -d '{"prompt":"Explain AI","model":"gemini","thinking_mode":false}'

# Run an Agent
curl http://your-host:3000/api/agent/run \
  -H "X-API-Key: lw_your_key_here" \
  -d '{"prompt":"Search for AI news","model":"gemini","tools":["web_search"]}'

# Get Analytics
curl http://your-host:3000/api/analytics/summary \
  -H "X-API-Key: lw_your_key_here"
